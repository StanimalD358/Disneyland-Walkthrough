        const landSelect = document.getElementById('land-select');

        const rideSelect = document.getElementById('ride-select');

        const itineraryList = document.getElementById('itinerary-list');

        const storyBox = document.getElementById('story-box');

        const displayImage = document.getElementById('display-image');

        const skipBtn = document.getElementById('skip-btn');


        // Audio System Variables

        const player1 = document.getElementById('player-1');

        const player2 = document.getElementById('player-2');

        let activePlayer = 1; // Tracks which player is currently audible

        let fadeInterval; // Used to manage the timing of the volume fade


        let itinerary = []; 

        let playbackQueue = []; 


        function updateRides() {

            const selectedLand = landSelect.value;

            rideSelect.innerHTML = '<option value="">-- Choose a Ride (Optional) --</option>';

            if (selectedLand && parkData[selectedLand]) {

                const rides = parkData[selectedLand].rides;

                for (const rideKey in rides) {

                    const option = document.createElement('option');

                    option.value = rideKey;

                    option.text = rides[rideKey].name;

                    rideSelect.appendChild(option);

                }

            }

        }


        function addToItinerary() {

            const landKey = landSelect.value;

            const rideKey = rideSelect.value;


            if (!landKey) {

                alert("Please select a land first!");

                return;

            }


            const landData = parkData[landKey];

            let displayText = landData.name;

            if (rideKey) displayText += " ➔ " + landData.rides[rideKey].name;


            const itemIndex = itinerary.length;

            itinerary.push({ land: landKey, ride: rideKey, index: itemIndex });


            const li = document.createElement('li');

            li.textContent = displayText;

            li.id = "itinerary-item-" + itemIndex; 

            itineraryList.appendChild(li);


            landSelect.value = "";

            updateRides();

        }


        function clearItinerary() {

            itinerary = [];

            playbackQueue = [];

            itineraryList.innerHTML = '';

            storyBox.innerHTML = 'Build your itinerary above!';

            displayImage.style.display = 'none';

            

            // Stop both players

            player1.pause();

            player2.pause();

            player1.src = '';

            player2.src = '';

            

            skipBtn.style.display = 'none';

        }


        function startWalkthrough() {

            if (itinerary.length === 0) {

                alert("Your itinerary is empty! Add some stops first.");

                return;

            }


            playbackQueue = []; 


            itinerary.forEach(item => {

                const landData = parkData[item.land];

                const randomAmbient = landData.ambientAudio[Math.floor(Math.random() * landData.ambientAudio.length)];

                

                playbackQueue.push({

                    audio: randomAmbient,

                    image: landData.image,

                    text: `Walking through ${landData.name}...`,

                    targetId: "itinerary-item-" + item.index

                });


                if (item.ride) {

                    const rideData = landData.rides[item.ride];

                    playbackQueue.push({

                        audio: rideData.audio,

                        image: rideData.image,

                        text: `Riding ${rideData.name}!`,

                        targetId: "itinerary-item-" + item.index

                    });

                }

            });


            const lastItem = itinerary[itinerary.length - 1];

            const lastLandData = parkData[lastItem.land];

            const finalAmbient = lastLandData.ambientAudio[Math.floor(Math.random() * lastLandData.ambientAudio.length)];

            

            playbackQueue.push({

                audio: finalAmbient,

                image: lastLandData.image,

                text: `Exiting back into ${lastLandData.name}...`,

                targetId: "itinerary-item-" + lastItem.index

            });


            skipBtn.style.display = 'block';

            playNextInQueue();

        }


        function skipStep() {

            playNextInQueue();

        }


        function playNextInQueue() {

            // Remove previous UI highlights

            const allItems = document.querySelectorAll('#itinerary-list li');

            allItems.forEach(li => li.classList.remove('active-item'));


            let currentPlayer = activePlayer === 1 ? player1 : player2;

            let nextPlayer = activePlayer === 1 ? player2 : player1;


            if (playbackQueue.length === 0) {

                storyBox.innerHTML = "Your magical day has come to an end!";

                displayImage.style.display = "none";

                skipBtn.style.display = 'none';

                

                // Fade out the last song to complete silence

                fadeOutSingle(currentPlayer);

                return;

            }


            const currentStep = playbackQueue.shift(); 


            // Highlight the new UI element

            if (currentStep.targetId) {

                const activeListItem = document.getElementById(currentStep.targetId);

                if (activeListItem) activeListItem.classList.add('active-item');

            }


            // Update text and image

            storyBox.innerHTML = currentStep.text;

            if (currentStep.image) {

                displayImage.src = currentStep.image;

                displayImage.style.display = 'block';

            } else {

                displayImage.style.display = 'none';

            }


            // --- THE CROSSFADE MAGIC ---

            nextPlayer.src = currentStep.audio;

            nextPlayer.volume = 0; // Start at zero volume

            

            nextPlayer.play().then(() => {

                // If a song is currently playing, crossfade them. Otherwise, just fade the first one in.

                if (!currentPlayer.paused && currentPlayer.src) {

                    crossfade(currentPlayer, nextPlayer);

                } else {

                    fadeInSingle(nextPlayer);

                }

                // Switch the active player tag for the next round

                activePlayer = activePlayer === 1 ? 2 : 1;

            }).catch(e => {

                console.log("Audio not found: " + currentStep.audio);

            });

        }


        // Fades one player OUT while fading the other IN over 2 seconds

        function crossfade(fadeOutPlayer, fadeInPlayer) {

            clearInterval(fadeInterval);

            let volume = 0;

            

            fadeInterval = setInterval(() => {

                volume += 0.05; // 5% volume change per tick

                

                if (volume >= 1) {

                    clearInterval(fadeInterval);

                    fadeInPlayer.volume = 1;

                    fadeOutPlayer.volume = 0;

                    fadeOutPlayer.pause(); // Turn off the old player to save resources

                } else {

                    fadeInPlayer.volume = volume;

                    fadeOutPlayer.volume = 1 - volume;

                }

            }, 100); // Ticks every 100 milliseconds

        }


        // Fades a single player IN (used for the very first track)

        function fadeInSingle(player) {

            clearInterval(fadeInterval);

            let volume = 0;

            player.volume = 0;

            

            fadeInterval = setInterval(() => {

                volume += 0.05;

                if (volume >= 1) {

                    clearInterval(fadeInterval);

                    player.volume = 1;

                } else {

                    player.volume = volume;

                }

            }, 100);

        }


        // Fades a single player OUT (used for the very end of the walkthrough)

        function fadeOutSingle(player) {

            clearInterval(fadeInterval);

            let volume = player.volume;

            

            fadeInterval = setInterval(() => {

                volume -= 0.05;

                if (volume <= 0) {

                    clearInterval(fadeInterval);

                    player.volume = 0;

                    player.pause();

                } else {

                    player.volume = volume;

                }

            }, 100);

        }


        // When a track finishes naturally, trigger the next one

        player1.onended = function() { playNextInQueue(); };

        player2.onended = function() { playNextInQueue(); };
